special_project = {
	key = "human_fallen_empire_pyramid_project"
	cost = 0
	days_to_research = 180
	tech_department = engineering_technology
	picture = GFX_evt_underground_civilization
	icon = "gfx/interface/icons/situation_log/situation_log_precursor.dds"
	
	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		skill > 1
	}
	
	event_scope = planet_event
	
	on_success = {
		if = {
			limit = {
				free_pop_tiles > 2
			}
			random_list = {
				33 = { planet_event = { id = colony.151 } }
				33 = { planet_event = { id = colony.152 } }
				33 = { planet_event = { id = colony.153 } }
			}
		}
		if = {
			limit = {
				NOT = { free_pop_tiles > 2 }
			}
			random_list = {
				50 = { planet_event = { id = colony.151 } }
				50 = { planet_event = { id = colony.153 } }
			}
		}		
	}
	
	on_fail = {
		
	}
}
